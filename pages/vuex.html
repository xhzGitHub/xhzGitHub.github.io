          # Vuex 的安装 (install)

          最开始的是完成对 Vuex 的安装，我们在平时引入 Vuex 的时候，通过使用 new Vuex.Store 创建一个 vuex 的实例，然后在根页面的 Vue 初始化时作为参数传进去：

          // main.js

import Vue from 'vue';
import App from './App.vue';
import store from './store';

Vue.config.productionTip = false

new Vue({
  store,
  render: h => h(App),
}).$mount('#app')

          // store/index.js

import Vue from 'vue';
import Vuex from 'vuex';

Vue.use(Vuex);

const state = {
  count: 0
};

const getters = {};

const mutations = {};

const actions = {};

export default new Vuex.Store({
  state,
  getters,
  mutations,
  actions,
})

          此处注意 Vue.use(Vuex), 这就是 Vue 对插件的安装方法，Vue 将本身实例作为参数传进去，并且执行了插件的 install 方法，那么我们本这个目标去实现一下安装。

          创建文件： vuex/index

          class Store {
  constructor(options) {
    install(Vue)
  }
}

function install(_Vue) {
  let Vue = _Vue;
  Vue.mixin({
    beforeCreate() {
      if (this.$options && this.$options.store) {
        this.$store = this.$options.store;
      } else {
        this.$store = this.$parent && this.$parent.$store;
      }
    }
  })
}

          其实 Vuex 的 install 方法就是在 Vue 实例中混入 beforeCreate 生命周期，此时需要在 Vue 实例下挂载 $store 属性，如果是根组件则直接取 new Vuex.Store(options) 传入的参数，如果不是根组件，则取父组件的该属性。

          # 创建 State 与 Getters

          需要在页面层显示 state 与 getters：

          <div id="app">
  【state】: {{ $store.state.count }}
  <div>
    【getters】: {{ $store.getters.countOfCake }}
  </div>
</div>

          那么我们需要在 Store 类中创建 state 与 getters：

          class Store {
  constructor(options) {
    install(Vue)

    this._state = options.state;
    this._getters = Object.create(null);

    resetStoreVM(this, this._state, options.getters);
  }

  get state() {
    return this._vm.$data.$$state;
  }

  get getters() {
    return this._getters;
  }
}

function resetStoreVM(store, state, getters) {
  Object.keys(getters).forEach(key => {
    Object.defineProperty(store._getters, key, {
      get: () => {
        return getters[key](store._state);
      }
    });
  });

  // 实现数据实时变化
  store._vm = new Vue({
    data: {
      $$state: state
    }
  })
}

          这里有个 resetStoreVM 的方法，目的是让 state 属性被 Vue 监听，当 set 的时候能够实现实时变化；另外就是去声明 getters，getters 的实现原理就是遍历传入的 optipons 的 getters，然后给 Store 的私有熟悉 _getters 创建相同 key 名的属性，并定义 get 方法，其值是 optipons.getters 方法执行后的返回值，这样就可以实现直接 this.$store.getters.xxx 可以直接访问到值。

          # mutations 和 actions

          接下里需要注册 mutations 和 actions 方法。

          首先是 mutations，遍历注册 options.mutations, 并且创建 commit 方法，执行对应 type 的 mutation:

          class Store {
  constructor(options) {
    this._mutations = Object.create(null);
    
    forEach(options.mutations, (type, mutation) => {
      this._mutations[type] = (payload) => {
        mutation(this._state, payload)
      };
    });
  }

  commit(_type, _payload) {
    this._mutations[_type](_payload);
  }
}

          这里用了一个 forEach 方法，因为后面对 actions 与 getters 的创建同样复用该方法。

          // utils.js

function forEach(object, cb) {
  Object.keys(object).forEach(key => {
    cb(key, object[key]);
  });
}

          再来是 actions，actions 通过 commit 一个 mutation 来实现间接改变状态的目的。

          这里 actions 与 mutations 的区别是 args 的区别，第一个参数需要将整个 store 作为参数传入：

          class Store {
  constructor(options) {
    this._actions = Object.create(null);

    forEach(options.actions, (type, action) => {
      this._actions[type] = (payload) => {
        action(this, payload);
      };
    });
  }

  dispatch(_type, _payload) {
    this._actions[_type](_payload);
  }
}

          基于以上，我们实现了最基础版的 vuex。但是当业务场景复杂起来，state 数量会很多，显得十分臃肿，vuex 的解决方案是「模块化」—— modules。

          接下来需要对 vuex 进行模块化封装。

          # 模块 Modules

          首先，在模块的使用上：

          const state = { name: 'xhz' };

const modules = {
  math: {
    state: {
      count: 5
    },
    getters,
    mutations,
    actions,
    modules: {
      college: {
        state: {
          name: 'son'
        }
      }
    }
  },
  chinese: {
    state: {
      teacher: 'chen'
    }
  }
};

Vue.use(Vuex);

export default new Vuex.Store({
  state,
  modules
})

          那么我们在打印 this.$store._modules，长这样：

          {
  root: {
    _rawModule: {...},
    _children: {
      math: {...},
      chinese: {...}
    }
    state: {...}
  }
}

          在这里 _rawModule 就是传进来的 options，_children 则是 modules 中的子模块。

          为了实现模块化，需要创建一个 ModuleCollection 类，这个类主要是输出以上的数据结构，并且实现递归。

          import { forEach } from './utils';

class ModuleCollection {
  constructor(options) {
    this.register([], options);
  }

  register(path, rootModule) {
    let module = {
      _rawModule: rootModule,
      _children: {},
      state: rootModule.state
    };

    // 根节点直接赋值
    if (!path.length) {
      this.root = module;
    } else {
      // 非根节点，寻找本轮父节点，针对父节点的 children 进行赋值
      let parent = path.slice(0, -1).reduce((root, cur) => {
        return root._children[cur];
      }, this.root);
      parent._children[path[path.length - 1]] = module;
    }
    
    // 递归
    if (rootModule.modules) {
      forEach(rootModule.modules, (key, module) => {
        this.register(path.concat(key), module);
      })
    }
  }
}

export default ModuleCollection;

          在实现了 ModuleCollection 类后，需要对之前 state, getters 等的注册进行重新封装：

          • 声明 store._modules 属性，作为 ModuleCollection 的实例；
          • installModule 方法负责重新注册模块的各 api；
          
class Store {
  constructor(options) {
    this._modules = new ModuleCollection(options);
    installModule(this, this._state, [], this._modules.root);
  }
}

function installModule(store, rootState, path, rootModule) {
  if (path.length > 0) {
      let parent = path.slice(0, -1).reduce((root, current) => {
          return root[current];
      }, rootState);

      // 响应式 
      Vue.set(parent, path[path.length-1], rootModule.state);
  }

  if (rootModule._raw.getters) {
      forEach(rootModule._raw.getters, (getterName, getterFn) => {
          Object.defineProperty(store.getters, getterName, {
              get: () => {
                  return getterFn(rootModule.state);
              }
          })
      })
  }

  if (rootModule._raw.actions) {
      forEach(rootModule._raw.actions, (actionName, actionFn) => {
          let entry = store.actions[actionName] || (store.actions[actionName] = []);
          entry.push(() => {
              actionFn.call(store, store)
          });
      })
  }

  if (rootModule._raw.mutations) {
      forEach(rootModule._raw.mutations, (mutationName, mutationFn) => {
          let entry = store.mutations[mutationName] || (store.mutations[mutationName] = []);
          entry.push(() => {
              mutationFn.call(store, rootModule.state)
          });
      })
  }

  // 递归
  forEach(rootModule._children, (childName, module) => {
      installModule(store, rootState, path.concat(childName), module);
  });
}

          这样，模块化的 vuex 就大功告成了。
